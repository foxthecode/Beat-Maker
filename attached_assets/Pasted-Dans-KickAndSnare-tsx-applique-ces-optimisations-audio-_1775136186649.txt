au scheduler de pattern, ni aux états React.
Applique dans l'ordre exact.

━━━ CORRECTION 3 — Limiter les noeuds simultanés ━━━

Sur mobile, le clap et le hihat créent trop de
noeuds simultanément. Dans _syn(), modifie les
synthèses concernées :

Pour le HIHAT, remplace les 6 oscillateurs par
4 sur mobile :
hihat: () => {
  const decay = 0.045 * sDec;
  const mg = gain(1); mg.connect(d);
  // Mobile : 4 oscillateurs au lieu de 6
  const freqs = this.isMobile
    ? [80, 167, 273, 329]
    : [80, 119, 167, 219, 273, 329];
  freqs.map(f => f * sTone).forEach(f => {
    const o = osc("square", f);
    const og = gain(v * 0.06);
    og.gain.exponentialRampToValueAtTime(
      0.001, t + decay);
    o.connect(og); og.connect(mg);
    o.start(t); o.stop(t + decay + 0.001);
  });
  const ns = noise(decay);
  const hp = filt("highpass", 8000 * sTone);
  const ng = gain(v * 0.18 * sSnap);
  ng.gain.exponentialRampToValueAtTime(
    0.001, t + decay);
  ns.connect(hp); hp.connect(ng); ng.connect(d);
  ns.start(t); ns.stop(t + decay + 0.001);
},

Pour le CLAP, remplace les 4 strates par
2 sur mobile :
clap: () => {
  const offsets = this.isMobile
    ? [0, 0.018]
    : [0, 0.009, 0.018, 0.028];
  offsets.map(off => off * sSnap).forEach(off => {
    const ns = noise(0.012);
    const bp = filt("bandpass", 1200, 1.5);
    const g = gain(v * 0.55 * sPunch);
    g.gain.setValueAtTime(0, t);
    g.gain.setValueAtTime(
      v * 0.55 * sPunch,
      t + Math.max(0.0001, off));
    g.gain.exponentialRampToValueAtTime(
      0.001, t + Math.max(0.001, off) + 0.012);
    ns.connect(bp); bp.connect(g); g.connect(d);
    ns.start(t + Math.max(0, off));
    ns.stop(t + Math.max(0, off) + 0.015);
  });
  const tailDur = 0.2 * sDec;
  const tail = noise(tailDur);
  const bp2 = filt("bandpass", 1000, 0.8);
  const tg = gain(v * 0.45 * sBody);
  tg.gain.setValueAtTime(0, t);
  tg.gain.setValueAtTime(v * 0.45 * sBody, t + 0.04);
  tg.gain.exponentialRampToValueAtTime(
    0.001, t + tailDur);
  tail.connect(bp2); bp2.connect(tg); tg.connect(d);
  tail.start(t + 0.04); tail.stop(t + tailDur + 0.01);
},

━━━ CORRECTION 4 — Prérendu parallèle ━━━

Dans _loadDefaults(), remplace la boucle
séquentielle actuelle par un rendu parallèle :
async _loadDefaults() {
  const durs = {
    kick:1.2, snare:0.28, hihat:0.1,
    clap:0.28, tom:0.65, ride:0.45,
    crash:1.6, perc:0.65
  };
  // Rendu parallèle : tous les sons en même temps
  await Promise.all(
    Object.entries(durs).map(async ([id, dur]) => {
      try {
        const sr = this.ctx.sampleRate;
        const oCtx = new OfflineAudioContext(
          1, Math.ceil(sr * dur), sr);
        this._syn(id, 0, 1, oCtx.destination, oCtx);
        this.buf[id] = await oCtx.startRendering();
      } catch(e) {
        console.warn("808 prerender failed:", id, e);
      }
    })
  );
  this.onReady?.();
}
